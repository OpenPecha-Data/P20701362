སྐྱིད་སྡུག་ལམ་འཁྱེར་གྱི་བློ་སྦྱོང་བཞུགས་སོ།། བླ་མ་རྣམས་ལ་ཕྱག་འཚལ་ལོ། །སྐྱིད་སྡུག་ལམ་ཁྱེར་གྱི་མན་ངག་འདི་བྱང་ཆུབ་ཏུ་སེམས་བསྐྱེད་ལ། སྐྱིད་པར་བྱུང་ན་སྐྱིད་མཁན་དེ་ལུས་དང་སེམས་གང་ཡིན་ངོས་ཡེར་གྱིས་བཟུང་། ལུས་བེམ་པོ་རོ་དང་འདྲ་སྟེ། སྐྱིད་མཁན་དུ་མ་གྲུབ། སེམས་སྟོང་པ་ནམ་མཁའ་དང་འདྲ་སྟེ། ཁོ་ཉིད་ཁོ་རང་གིས་སྟོང་པས་སྐྱིད་པོར་མ་གྲུབ། དེས་ན་སྐྱིད་པ་པོར་འཛིན་པ་དེ་ཉིད་ཁོ་རང་གིས་སྟོང་པས། སྟོང་པ་ལ་སྟོང་པར་ཤེས་པར་བྱས་ལ་སྟོང་པའི་ངང་ལ་སིང་ངེ་ཡེར་རེ་ལྷོད་དེ་ཤིག་གེ་སྡོད་ཚད་མཉམ་པར་བཞག །དེ་ནི་མ་བཅོས་དྲི་མེད་ཆོས་སྐུ་ཐོབ་པའི་ཐབས་ཡིན་ནོ། །དེའི་ངང་ལ་སྐྱིད་པར་སྣང་བ་དེ་ཉིད་ཀྱི་ངོ་བོར་ཡེར་གྱིས་བསྒོམ། སེམས་ཡེངས་རུང་མ་ཡེངས་རུང་ཡི་དམ་ལྷ་དང་འབྲལ་མེད་དུ་རྒྱས་གདབ། དེ་ནི་དག་པའི་གདུལ་བྱ་འདུལ་བྱེད་ལོངས་སྐུ་ཐོབ་པའི་ཐབས་ཡིན་ནོ། །དེའི་ང་རྒྱལ་དང་མ་བྲལ་བས་སྐྱིད་པར་སྣང་བ་དེ་ཉིད་སེམས་ཅན་གྱི་དོན་དུ་བསྔོ་སྟེ། སྐྱིད་ན་དགེ་བ་ཚོགས་སུ་བསྔོ། །ཕན་བདེས་ནམ་མཁའ་གང་བར་ཤོག །ཅེས་ལན་གསུམ་བརྗོད། དེ་ནི་མ་དག་པའི་གདུལ་བྱ་འདུལ་བྱེད་སྤྲུལ་སྐུ་ཐོབ་པའི་ཐབས་ཡིན་ནོ། །སྡུག་མཁན་དེ་ལུས་དང་སེམས་གང་ཡིན་ངོས་ཡེར་གྱིས་བཟུང་། ལུས་བེམ་པོ་རོ་དང་འདྲ་སྟེ། སྡུག་མཁན་དུ་མ་གྲུབ། སེམས་སྟོང་པ་ནམ་མཁའ་དང་འདྲ་སྟེ། ཁོ་ཉིད་ཁོ་རང་གིས་སྟོང་པས་སྡུག་པ་པོར་མ་གྲུབ་པས། དེ་ནས་སྡུག་པར་འཛིན་པ་དེ་ཉིད་ཁོ་རང་གིས་སྟོང་པ་ལ་སྟོང་པར་བྱས་ལ་སྟོང་པའི་ངང་ལས་སིང་ངེ་ཡེར་རེ་ལྷོད་དེ་ཤིག་གེ་སྡོད་ཚད་དུ་མཉམ་པར་བཞག །མ་བཅོས་དྲི་མེད་ཆོས་སྐུ་ཐོབ་པའི་ཐབས་ཡིན་ནོ། །དེའི་ངང་ལ་སྡུག་པར་འཛིན་པ་དེ་ཉིད་རང་གི་ཡི་དམ་གང་ཡིན་པ་དེའི་ངོ་བོར་ཡེར་གྱིས་བསྒོམ། སེམས་ཡེངས་རུང་མ་ཡེངས་རུང་ཡི་དམ་ལྷ་དང་འབྲལ་མེད་དུ་རྒྱས་གདབ། དེ་ནི་དག་པའི་གདུལ་བྱ་འདུལ་བྱེད་ལོངས་སྐུ་ཐོབ་པའི་ཐབས་ཡིན་ནོ།
